function getDateDiff($date1, $date2)
{
    $datetime1 = new \DateTime($date1);
    $datetime2 = new \DateTime($date2);
    $interval = $datetime1->diff($datetime2);

    $datetime = [];
    $items = [
        'year'   => 'y',

        'day'    => 'd',
        'hour'   => 'H',
        'minute' => 'i',

    ];
    foreach ($items as $key => $format) {
        $datetime[$key] = $interval->format("%{$format}");
    }

    return $datetime;
}
