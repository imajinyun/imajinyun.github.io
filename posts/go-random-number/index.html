package main

import (
  "fmt"
  "math/rand"
  "time"
)

func init() {
  rand.Seed(time.Now().UnixNano())
}

func main() {
  for i := 0; i < 10; i++ {
    fmt.Println(randomString(10))
  }
}

func randomString(size int) string {
  bytes := make([]byte, size)
  for i := 0; i < size; i++ {
    bytes[i] = byte(randInt(97, 122))
  }

  return string(bytes)
}

func randInt(min, max int) int {
  return min + rand.Intn(max-min+1)
}
