    public static function sort(array &$array): array
    {
        $count = count($array);

        if ($count <= 1) {
            return $array;
        }

        $pivot = $array[0]; // å–æ•°ç»„ç¬¬ä¸€ä¸ªå…ƒç´ ä¸ºåŸºå‡†å€¼

        $right = self::sort($right);

        return array_merge($left, (array)$pivot, $right);
