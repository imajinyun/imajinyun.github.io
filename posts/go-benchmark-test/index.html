      if nums[j] < nums[j-1] {
        nums[j], nums[j-1] = nums[j-1], nums[j]
      }
    }
  }
}

func BenchmarkBubbleSort(b *testing.B) {
  for n := 0; n < b.N; n++ {
    b.StopTimer()
    nums := genWithCap(10000)
    b.StartTimer()
    bubbleSort(nums)
  }
}
