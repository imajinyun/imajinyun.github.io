<?php

class Foo
{
    public function getName()
    {
        return __METHOD__;
    }

    public function __call($method, $arguments)
    {
        if (method_exists($this, $method)) {
            return call_user_func_array([$this, $method], $arguments);
        }

        return null;
    }
}
