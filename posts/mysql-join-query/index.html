SELECT
  a.`id`    AS courseId,
  a.`title` AS courseTitle,
  a.`englishName`,
  b.`id`    AS lessonId,
  b.`title` AS lessonTitle,
  b.`tags`
FROM
  test.`course` AS a
  INNER JOIN test.`lesson` AS b ON a.`id` = b.`courseId`;
